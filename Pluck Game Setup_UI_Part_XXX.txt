    private void ShowLeaderboard(object sender, EventArgs e)
    {
        string leaderboardText = "Battle Pass Leaderboard:\n";
        foreach (var player in playerRankings.OrderByDescending(r => r.Value))
        {
            leaderboardText += $"{player.Key}: Level {player.Value}\n";
        }
        MessageBox.Show(leaderboardText, "Leaderboard", MessageBoxButtons.OK, MessageBoxIcon.Information);
    }

    private string GetAIEmotion()
    {
        Random rnd = new Random();
        var keys = new List<string>(aiEmotions.Keys);
        return aiEmotions[keys[rnd.Next(keys.Count)]];
    }

    private void DisplayAIEmotion()
    {
        string emotion = GetAIEmotion();
        gameLog.Text += "\nAI: " + emotion;
    }
