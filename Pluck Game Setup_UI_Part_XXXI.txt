    private void OptimizeNetworking()
    {
        server = new TcpListener(IPAddress.Any, Port);
        server.Start();
        Task.Run(() => AcceptClients());
    }

    private void AcceptClients()
    {
        while (connectedClients.Count < 6)
        {
            TcpClient client = server.AcceptTcpClient();
            connectedClients.Add(client);
        }
    }

    private void UpdateAILearning(string ai, int outcome)
    {
        if (!aiLearning.ContainsKey(ai)) aiLearning[ai] = 1200;
        aiLearning[ai] += outcome;
        gameLog.Text += $"\n{ai} adjusted learning score to {aiLearning[ai]}";
    }

    private void AdaptAIPlaystyle()
    {
        foreach (var ai in aiLearning.Keys.ToList())
        {
            int decisionAdjustment = aiLearning[ai] > 1300 ? 10 : -10;
            gameLog.Text += $"\n{ai} is adapting strategies based on performance.";
        }
    }
