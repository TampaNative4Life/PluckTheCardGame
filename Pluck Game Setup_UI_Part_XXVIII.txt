    private void ShowDailyChallenge(object sender, EventArgs e)
    {
        MessageBox.Show($"Today's Challenge: {currentDailyChallenge}", "Daily Challenge", MessageBoxButtons.OK, MessageBoxIcon.Information);
    }

    private void GenerateDailyChallenge()
    {
        string[] challenges = new string[]
        {
            "Win a match without losing a trick",
            "Pluck an opponent 3 times in a single game",
            "Win 2 ranked matches in a row",
            "Score the highest trick count in a match"
        };
        Random rnd = new Random();
        currentDailyChallenge = challenges[rnd.Next(challenges.Length)];
    }
