    private void StartTournament(object sender, EventArgs e)
    {
        isTournamentMode = true;
        gameLog.Text += "\nTournament Mode Started! Players will compete in elimination rounds.";
        tournamentRounds.Enqueue(new List<string> { "Player 1", "Player 2", "Player 3", "Player 4", "Player 5", "Player 6" });
        NextTournamentRound();
    }

    private void NextTournamentRound()
    {
        if (tournamentRounds.Count == 1)
        {
            gameLog.Text += "\nTournament Winner: " + tournamentRounds.Dequeue()[0];
            isTournamentMode = false;
            return;
        }
        
        var currentRound = tournamentRounds.Dequeue();
        gameLog.Text += "\nStarting Next Round with Players: " + string.Join(", ", currentRound);
        List<string> winners = SimulateTournamentRound(currentRound);
        tournamentRounds.Enqueue(winners);
    }

    private List<string> SimulateTournamentRound(List<string> players)
    {
        Random rnd = new Random();
        List<string> winners = new List<string>();
        for (int i = 0; i < players.Count / 2; i++)
        {
            string winner = players[rnd.Next(players.Count)];
            winners.Add(winner);
            gameLog.Text += "\nMatch: " + players[i * 2] + " vs " + players[i * 2 + 1] + " - Winner: " + winner;
        }
        return winners;
    }
