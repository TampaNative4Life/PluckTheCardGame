    private void SaveAIMemory(string player, int successRate)
    {
        aiMemory[player] = successRate;
    }

    private int GetAIPastPerformance(string player)
    {
        return aiMemory.ContainsKey(player) ? aiMemory[player] : 50; // Default to 50% performance if no memory
    }

    private string GetAIMove(string player)
    {
        int aiPerformance = GetAIPastPerformance(player);
        return game.GetSmartAIMove(player, aiPerformance);
    }
