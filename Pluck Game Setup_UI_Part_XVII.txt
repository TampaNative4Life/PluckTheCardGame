    private void HostGame(object sender, EventArgs e)
    {
        server = new TcpListener(IPAddress.Any, Port);
        server.Start();
        stream = server.AcceptTcpClient().GetStream();
        isHost = true;
        MessageBox.Show("Game Hosted. Waiting for players...", "Multiplayer", MessageBoxButtons.OK, MessageBoxIcon.Information);
    }

    private void JoinGame(object sender, EventArgs e)
    {
        client = new TcpClient("127.0.0.1", Port);
        stream = client.GetStream();
        MessageBox.Show("Connected to Host!", "Multiplayer", MessageBoxButtons.OK, MessageBoxIcon.Information);
    }
